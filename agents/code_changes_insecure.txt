+import os
+import sqlite3
+from flask import Flask, request, make_response
+
+app = Flask(__name__)

+DB_PATH = "users.db"
+API_SECRET = "hardcoded-dev-secret"  # hardcoded secret
+
+
+def get_db():
+    return sqlite3.connect(DB_PATH)
+
+
+@app.route("/login", methods=["POST"])
+def login():
+    # SQL injection: unsanitized string formatting
+    username = request.form.get("username", "")
+    password = request.form.get("password", "")
+    conn = get_db()
+    cur = conn.cursor()
+    cur.execute(f"SELECT * FROM users WHERE username = '{username}' AND password = '{password}'")
+    row = cur.fetchone()
+    if row:
+        resp = make_response({"status": "ok"})
+        # Insecure cookie: no HttpOnly/Secure, static secret
+        resp.set_cookie("session", API_SECRET)
+        return resp
+    return {"error": "invalid creds"}, 401
